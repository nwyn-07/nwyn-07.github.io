Fetch API và AJAX - Giao tiếp với Server

## Fetch API cơ bản // GET request fetch('https://api.example.com/users') &nbsp; .then(response => { &nbsp; if (!response.ok) { &nbsp; throw new Error('Network response was not ok'); &nbsp; } &nbsp; return response.json(); &nbsp; }) &nbsp; .then(data => { &nbsp; console.log('Users:', data); &nbsp; }) &nbsp; .catch(error => { &nbsp; console.error('Error:', error); &nbsp; }); \## POST request ```javascript // POST request với JSON fetch('https://api.example.com/users', { &nbsp; method: 'POST', &nbsp; headers: { &nbsp; 'Content-Type': 'application/json', &nbsp; }, &nbsp; body: JSON.stringify({ &nbsp; name: 'John Doe', &nbsp; email: 'john@example.com' &nbsp; }) }) .then(response => response.json()) .then(data => console.log('Success:', data)) .catch(error => console.error('Error:', error)); \## Async/Await với Fetch ```javascript async function fetchUser(userId) { &nbsp; try { &nbsp; const response = await fetch(`https://api.example.com/users/${userId}`); &nbsp; &nbsp; if (!response.ok) { &nbsp; throw new Error(`HTTP error! status: ${response.status}`); &nbsp; } &nbsp; &nbsp; const user = await response.json(); &nbsp; return user; &nbsp; } catch (error) { &nbsp; console.error('Failed to fetch user:', error); &nbsp; return null; &nbsp; } } // Sử dụng const user = await fetchUser(123); if (user) { &nbsp; console.log('User data:', user); } \## Upload file với Fetch ```javascript async function uploadFile(file) { &nbsp; const formData = new FormData(); &nbsp; formData.append('file', file); &nbsp; formData.append('description', 'Profile picture'); &nbsp; try { &nbsp; const response = await fetch('/api/upload', { &nbsp; method: 'POST', &nbsp; body: formData &nbsp; }); &nbsp; if (response.ok) { &nbsp; const result = await response.json(); &nbsp; console.log('Upload successful:', result); &nbsp; return result; &nbsp; } else { &nbsp; throw new Error('Upload failed'); &nbsp; } &nbsp; } catch (error) { &nbsp; console.error('Upload error:', error); &nbsp; throw error; &nbsp; } } \## Error Handling chi tiết ```javascript async function robustFetch(url, options = {}) { &nbsp; try { &nbsp; const response = await fetch(url, options); &nbsp; &nbsp; // Kiểm tra HTTP status &nbsp; if (response.status === 404) { &nbsp; throw new Error('Resource not found'); &nbsp; } &nbsp; &nbsp; if (response.status === 401) { &nbsp; throw new Error('Unauthorized - please login'); &nbsp; } &nbsp; &nbsp; if (response.status === 500) { &nbsp; throw new Error('Server error'); &nbsp; } &nbsp; &nbsp; if (!response.ok) { &nbsp; throw new Error(`HTTP error! status: ${response.status}`); &nbsp; } &nbsp; &nbsp; const data = await response.json(); &nbsp; return { success: true, data }; &nbsp; &nbsp; } catch (error) { &nbsp; console.error('Fetch failed:', error); &nbsp; return { &nbsp; success: false, &nbsp; error: error.message, &nbsp; status: error.name === 'TypeError' ? 'NETWORK\_ERROR' : 'HTTP\_ERROR' &nbsp; }; &nbsp; } } \## So sánh với XMLHttpRequest ```javascript // Cũ: XMLHttpRequest const xhr = new XMLHttpRequest(); xhr.open('GET', '/api/data'); xhr.onload = function() { &nbsp; if (xhr.status === 200) { &nbsp; console.log(JSON.parse(xhr.responseText)); &nbsp; } }; xhr.send(); // Mới: Fetch API fetch('/api/data') &nbsp; .then(response => response.json()) &nbsp; .then(data => console.log(data)); Kết luận: Fetch API cung cấp interface hiện đại, mạnh mẽ và dễ sử dụng cho HTTP requests.
